ULCL Dynamic
Traffic Steering

This guide demonstrates how to implement Dynamic Traffic Steering in a 5G Core network using the Uplink Classifier (ULCL). With the free5GC framework, it demonstrates how to route user traffic across different Anchor UPFs efficiently, adapting to real-time requirements.

Part of the network topology implementation, we used this guide by s5uishida.
It describes a simple network topology that uses free5GC and UERANSIM for ULCL with one I-UPF and two PSA-UPFs. We used go-upf for the implementation of the UPFs.

free5GC v3.3.0 UERANSIM gtp5g v0.8.6

Dynamic Steering API

REST endpoint on the SMF that triggers live path switching via PFCP Session Modification.

api_steering.go

gosbi/api_steering.go 
package sbi

import (
    "net/http"
    "github.com/gin-gonic/gin"
)

func (s *Server) getSteeringRoutes() []Route {
    return []Route{
        { Name: "Steer Traffic", Method: http.MethodPost, Pattern: "/steer", APIFunc: s.HTTPSteerTraffic },
    }
}

func (s *Server) HTTPSteerTraffic(c *gin.Context) {
    s.Processor().HandleSteerTraffic(c)
}

steering.go

ⓘ
New file — created from scratch Lives at NFs/smf/internal/sbi/processor/steering.go. Implements HandleSteerTraffic, the handler wired up by api_steering.go above. Contains two key bug-fixes required for multi-UE operation — see § 16 for full diffs and root-cause analysis.
goNFs/smf/internal/sbi/processor/steering.go 
package processor

import (
    "fmt"
    "net/http"
    "github.com/free5gc/openapi/models"
    pfcpType "github.com/free5gc/pfcp/pfcpType"
    "github.com/free5gc/smf/internal/context"
    smf_context "github.com/free5gc/smf/internal/context"
    "github.com/gin-gonic/gin"
)

type SteerRequest struct {
    SUPI      string `json:"supi"`
    PDUSessID int32  `json:"pduSessId"`
    TargetUPF string `json:"targetUpf"`
}

func (p *Processor) HandleSteerTraffic(c *gin.Context) {
    var req SteerRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    if req.PDUSessID == 0 { req.PDUSessID = 1 }

    smContext := smf_context.GetSMContextById(req.SUPI, req.PDUSessID)
    if smContext == nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "SMContext not found", "supi": req.SUPI})
        return
    }

    upi := smf_context.GetUserPlaneInformation()
    targetUPNode := upi.GetUPFNodeByIP(req.TargetUPF)
    if targetUPNode == nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "Target UPF not found", "targetUpf": req.TargetUPF})
        return
    }
    targetUPF := targetUPNode.UPF
    targetUPFName := upi.GetUPFNameByIp(req.TargetUPF)

    smContext.SMLock.Lock()
    defer smContext.SMLock.Unlock()

    iface := targetUPF.GetInterface(models.UpInterfaceType_N9, smContext.Dnn)
    if iface == nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Target UPF has no N9 interface"})
        return
    }
    targetIP, err := iface.IP(smContext.SelectedPDUSessionType)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": fmt.Sprintf("Cannot get target UPF IP: %v", err)})
        return
    }

    var ulcfNodes []*context.DataPathNode
    for _, dp := range smContext.Tunnel.DataPathPool {
        if dp.IsDefaultPath { continue }
        for node := dp.FirstDPNode; node != nil; node = node.Next() {
            if node.UpLinkTunnel == nil || node.UpLinkTunnel.PDR == nil { continue }
            pdr := node.UpLinkTunnel.PDR
            if pdr.PDI.SDFFilter == nil || pdr.FAR == nil ||
               pdr.FAR.ForwardingParameters == nil ||
               pdr.FAR.ForwardingParameters.OuterHeaderCreation == nil { continue }
            ulcfNodes = append(ulcfNodes, node)
        }
    }
    if len(ulcfNodes) == 0 {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "no ULCL steering FAR found pointing to old UPF"})
        return
    }

    var pdrList []*context.PDR
    var farList []*context.FAR
    debugFARs := ""
    for _, node := range ulcfNodes {
        ulFAR := node.UpLinkTunnel.PDR.FAR
        if ulFAR == nil || ulFAR.ForwardingParameters == nil { continue }
        existingTEID := uint32(0)
        if ulFAR.ForwardingParameters.OuterHeaderCreation != nil {
            existingTEID = ulFAR.ForwardingParameters.OuterHeaderCreation.Teid
        }
        ulFAR.ForwardingParameters.OuterHeaderCreation = &pfcpType.OuterHeaderCreation{
            OuterHeaderCreationDescription: pfcpType.OuterHeaderCreationGtpUUdpIpv4,
            Ipv4Address:                    targetIP,
            Teid:                           existingTEID,
        }
        ulFAR.State = smf_context.RULE_UPDATE
        pdrList = append(pdrList, node.UpLinkTunnel.PDR)
        farList = append(farList, ulFAR)
        debugFARs += fmt.Sprintf("FARID=%d,UPF=%s;",
            ulFAR.FARID, node.UPF.NodeID.ResolveNodeIdToIp().String())
    }
    if len(farList) == 0 {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "no valid FARs to update"})
        return
    }
    c.Header("X-Debug-FAR", debugFARs)

    iUPFNode := ulcfNodes[0]
    resChan := make(chan SendPfcpResult, 1)
    pfcpState := &PFCPState{
        upf:     iUPFNode.UPF,
        pdrList: pdrList,
        farList: farList,
        barList: []*context.BAR{},
        qerList: nil,
        urrList: nil,
    }
    go modifyExistingPfcpSession(smContext, pfcpState, resChan, "")
    result := <-resChan
    close(resChan)

    if result.Err != nil || result.Status != smf_context.SessionUpdateSuccess {
        c.JSON(http.StatusInternalServerError, gin.H{
            "error":  fmt.Sprintf("PFCP modification failed: %v", result.Err),
            "status": result.Status,
        })
        return
    }
    smContext.SelectedUPF = targetUPNode
    c.JSON(http.StatusOK, gin.H{
        "status":    "steering applied",
        "supi":      req.SUPI,
        "targetUpf": targetUPFName,
        "targetIp":  req.TargetUPF,
    })
}

Troubleshooting

SMContext not found — curl returns 404
The UE is not connected. Reconnect it.
"No PDR match this skb" in dmesg
The gtp5g kernel patch was not applied or the old module is still loaded.
"file exists" error on UPF start
sudo ip link delete upfgtp

Verified implementationLive bidirectional switching (PSA-UPF ↔ PSA-UPF2) with no session interruption confirmed on free5GC v3.3.0 with gtp5g v0.8.6.
